    ${headers}
    ${sources}
    )

    ${target}
    )

    PROPERTIES
    ${DEFAULT_PROJECT_OPTIONS}
    )

target_include_directories(${target}
    PRIVATE

    PUBLIC
    ${DEFAULT_INCLUDE_DIRECTORIES}
    ${CMAKE_CURRENT_SOURCE_DIR}/source
    ${CMAKE_CURRENT_SOURCE_DIR}/source/OrbSlam

    ${g2o_INCLUDE_DIR}

    ${WAI_ROOT}/thirdparty/DBoW2
    ${WAI_ROOT}/thirdparty/utils/lib-Utils/source

    )

    PRIVATE

    PUBLIC
    ${DEFAULT_COMPILE_DEFINITIONS}
    WAI_DLL
    WAI_DLL_EXPORTS

    INTERFACE
    )

target_compile_options(${target}
    PRIVATE

    PUBLIC
    ${DEFAULT_COMPILE_OPTIONS}

    INTERFACE
    )

message(STATUS "PlatformLinkLibs ${PlatformLinkLibs}")
target_link_libraries(${target}
    PRIVATE

    PUBLIC
    ${PlatformLinkLibs}
    ${DEFAULT_LINKER_OPTIONS}
    ${OpenCV_LIBS}
    ${g2o_LIBS}
    ${META_PROJECT_NAME}::lib-DBoW2
    ${META_PROJECT_NAME}::lib-Utils
